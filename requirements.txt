import streamlit as st
import requests
import random

# 1. API Ayarları
API_KEY = "3351c09db12feaab6b5b1e1673e7376c"
BASE_URL = "https://api.themoviedb.org/3"
DISNEY_COMPANY_ID = "2" # Walt Disney Pictures TMDB ID'si

# Web Sitesi Tasarım Ayarları
st.set_page_config(page_title="MoodStream", page_icon="🎬", layout="centered")

st.title("🎬 MoodStream")
st.subheader("Bugün hangi moddasın? Sana en uygun filmi bulalım!")
st.write("---")

# Sayfalar arası veri taşımak için session_state ayarları
if "step" not in st.session_state:
    st.session_state.step = 1
    st.session_state.puanlar = {16: 0, 28: 0, 35: 0, 18: 0, 99: 0, 27: 0, 10749: 0}
    st.session_state.film_index = 0

# --- BENZER YAPIMLARI GETİRME FONKSİYONU ---
def benzer_filmleri_getir(film_id, medya_turu):
    recommend_url = f"{BASE_URL}/{medya_turu}/{film_id}/recommendations?api_key={API_KEY}&language=tr-TR"
    try:
        response = requests.get(recommend_url).json()
        benzerler = response.get('results', [])
        if benzerler:
            st.write("💡 **Eğer bu tarz yapımları seviyorsan, listene ekleyebileceğin 3 benzer öneri:**")
            for i, b_film in enumerate(benzerler[:3]):
                baslik = b_film.get('title') if medya_turu == "movie" else b_film.get('name')
                st.write(f"   {i+1}. {baslik} (Puanı: {b_film['vote_average']})")
    except:
        pass

# --- WEB ARAYÜZÜ VE SENİN ÖZGÜN SORULARIN ---
if st.session_state.step == 1:
    st.markdown("### 1) Şu an tam olarak nasıl hissediyorsun?")
    if st.button("A) Enerjik, yerimde duramıyorum"):
        st.session_state.puanlar[28] += 2
        st.session_state.step = 2
        st.rerun()
    if st.button("B) Yorgun, kafa dinlemek istiyorum"):
        st.session_state.puanlar[99] += 2
        st.session_state.step = 2
        st.rerun()
    if st.button("C) Mutsuz, gün biraz fazla yordu"):
        st.session_state.puanlar[18] += 2
        st.session_state.step = 2
        st.rerun()

elif st.session_state.step == 2:
    st.markdown("### 2) Mutluluk senin için hangi renk?")
    if st.button("A) Sarı / Turuncu (Cıvıl cıvıl)"):
        st.session_state.puanlar[35] += 2  # Komedi
        st.session_state.puanlar[16] += 2  # Animasyon
        st.session_state.step = 3
        st.rerun()
    if st.button("B) Mavi / Gri (Sakin ve derin)"):
        st.session_state.puanlar[18] += 2
        st.session_state.step = 3
        st.rerun()
    if st.button("C) Yeşil / Kırmızı"):
        st.session_state.puanlar[10749] += 2
        st.session_state.step = 3
        st.rerun()

elif st.session_state.step == 3:
    st.markdown("### 3) Gün içindeki mood değişimi nasıldı?")
    if st.button("A) Çok dalgalıydı, bir öyle bir böyleydim"):
        st.session_state.puanlar[10749] += 2
        st.session_state.step = 4
        st.rerun()
    if st.button("B) Gayet sabitti, sakindi"):
        st.session_state.puanlar[27] += 2
        st.session_state.step = 4
        st.rerun()

elif st.session_state.step == 4:
    st.markdown("### 4) Hangisi sana daha çok hitap ediyor?")
    if st.button("A) Gülerken ağlamalık hikayeler"):
        st.session_state.puanlar[18] += 2
        st.session_state.puanlar[35] += 1
        st.session_state.step = 5
        st.rerun()
    if st.button("B) Ağlarken gülmelik (Kara mizah) hikayeler"):
        st.session_state.puanlar[35] += 2
        st.session_state.puanlar[18] += 1
        st.session_state.step = 5
        st.rerun()

elif st.session_state.step == 5:     
    st.markdown("### 5) Hangisi bugün en çok yapmak istediğindi?")   
    if st.button("A) Eve hemen dönüp uyumak"):
        st.session_state.puanlar[16] += 2
        st.session_state.puanlar[10749] += 2
        st.session_state.step = 6
        st.rerun()
    if st.button("B) TÜM gün dışarda geçirmek"):
        st.session_state.puanlar[35] += 2
        st.session_state.puanlar[18] += 1
        st.session_state.step = 6
        st.rerun()   

# --- SONUÇ EKRANI ---
elif st.session_state.step == 6:
    secilen_tur = max(st.session_state.puanlar, key=st.session_state.puanlar.get)
    
    medya = "movie"
    extra_filters = ""
    with_genres = str(secilen_tur)
    kategori_adi = "Genel Havuz"

    # Güvenlik ve kalite kalkanı (Tanımlama)
    kalite_kalkani = "&include_adult=false&vote_count.gte=200"

    # Disney Animasyon Filtresi
    if secilen_tur == 16:
        extra_filters = f"&with_companies={DISNEY_COMPANY_ID}"
        kategori_adi = "✨ Walt Disney Animasyon Klasikleri"

    # Romantik Komedi & Dönem Dizisi Kontrolü
    elif (secilen_tur == 35 or secilen_tur == 10749) and st.session_state.puanlar[35] > 0 and st.session_state.puanlar[10749] > 0:
        if "romantik_mod" not in st.session_state:
            st.session_state.romantik_mod = random.choice(["romedi_filmleri", "donem_dizileri"])
            
        if st.session_state.romantik_mod == "romedi_filmleri":
            medya = "movie"
            with_genres = "35,10749"
            extra_filters = "&with_original_language=tr|en"
            kategori_adi = "💖 Seçkin TR, US & UK Romantik Komedileri"
        else:
            medya = "tv"
            with_genres = "10749,18,36"
            extra_filters = "&with_original_language=en"
            kategori_adi = "👑 Bridgerton & Outlander Tarzı Tarihi Dönem Dizileri"
            
    else:
        if secilen_tur == 28: kategori_adi = "⚡ Dinamik Aksiyon Yapımları"
        elif secilen_tur == 27: kategori_adi = "👁️ Gerilim & Gizem Dünyası"
        elif secilen_tur == 18: kategori_adi = "🎭 Derin ve Anlamlı Dramalar"
        elif secilen_tur == 99: kategori_adi = "🧠 Ufuk Açıcı Belgeseller"

    # Veriyi Çekme
    if "filmler" not in st.session_state:
        try:
            rastgele_sayfa = random.randint(1, 2)
            # BURADAKİ YAZIM HATASI DÜZELTİLDİ: {kalite_kalkani} yapıldı
            discover_url = f"{BASE_URL}/discover/{medya}?api_key={API_KEY}&with_genres={with_genres}{extra_filters}{kalite_kalkani}&sort_by=popularity.desc&language=tr-TR&page={rastgele_sayfa}"
            response = requests.get(discover_url, timeout=5).json()
            st.session_state.filmler = response.get('results', [])
            random.shuffle(st.session_state.filmler)
        except:
            st.error("📡 Sunucuya bağlanırken ufak bir kesinti oldu. Lütfen internetini kontrol edip sayfayı yenile.")
            st.session_state.filmler = []

    filmler = st.session_state.filmler

    if filmler and st.session_state.film_index < len(filmler):
        film = filmler[st.session_state.film_index]
        film_id = film['id']
        baslik = film.get('title') if medya == "movie" else film.get('name')
        
        st.caption(f"Hedeflenen Tarz: {kategori_adi}")
        st.success(f"🎬 **MoodStream Önerisi:** {baslik}")
        st.write(f"📝 **Özet:** {film['overview'] if film['overview'] else 'Türkçe özet bulunamadı.'}")
        
        # Fragman
        fragman_link = None
        try:
            video_url = f"{BASE_URL}/{medya}/{film_id}/videos?api_key={API_KEY}"
            video_response = requests.get(video_url, timeout=3).json()
            for video in video_response.get('results', []):
                if video['site'] == 'YouTube':
                    fragman_link = f"https://www.youtube.com/watch?v={video['key']}"
                    break
        except:
            pass
        
        if fragman_link:
            st.video(fragman_link)
        else:
            st.warning("🍿 Bu yapım için uygun bir YouTube fragmanı yüklenemedi.")
            
        col1, col2 = st.columns(2)
        with col1:
            if st.button("🎉 Beğendim, bunu izleyeceğim!"):
                st.balloons()
                benzer_filmleri_getir(film_id, medya)
        with col2:
            if st.button("👎 Beğenmedim, başka bir şey göster"):
                st.session_state.film_index += 1
                st.rerun()
                
    else:
        st.info("Bu havuzdaki tüm seçenekleri inceledin.")
                
    if st.button("🔄 Testi Yeniden Başlat"):
        st.session_state.clear()
        st.rerun()

    streamlit
    requests
